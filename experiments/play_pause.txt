running = True
def Run(b):
    global running, remember_dt,dt
    running = not running
    if running:
        b.text = "Pause"
        dt = remember_dt
    else:
        b.text = "Run"
        remember_dt = dt
        dt =0
    return

button(text="Pause",pos=scene.title_anchor,bind=Run)
